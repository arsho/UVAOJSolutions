#include <bits/stdc++.h>
using namespace std;
int main()
{
        ios_base::sync_with_stdio(0);
        cin.tie(0);
        int cas,t;
        long long n,ans;
        cin>>cas;
        for(t=1;t<=cas;t++)
        {
                cin>>n;
                ans=n/2;
                cout<<ans<<endl;
        }
        return 0;
}
