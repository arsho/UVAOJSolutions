{
    return x*x;
}

{
    int b,j,a,sho,i;
    long long ans ;
    scanf("%d", &b);

    {
        scanf("%d",&a);
        if(a<=0)

        {
            ans= 0;
            sho=sqrt(a);
            for(i=1 ; i<=sho; i++)
            {
                ans += a / i;
            }
            ans = 2*ans - sqr((int)sqrt(a));

        }
    }

}
